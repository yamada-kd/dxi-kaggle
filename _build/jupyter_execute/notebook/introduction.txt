#!/usr/bin/env python3
import os
import json

def main():
    fin = open("/content/drive/MyDrive/kaggle/token/kaggle.json", "r")
    json_data = json.load(fin)
    fin.close()
    os.environ["KAGGLE_USERNAME"] = json_data["username"]
    os.environ["KAGGLE_KEY"] = json_data["key"]

if __name__ == "__main__":
    main()
